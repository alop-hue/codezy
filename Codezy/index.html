أمثلة تفاعلية

جرب الكود بنفسك وشاهد النتيجة فوراً

مثال: التحقق من الأعداد الأولية 
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

# اختبار الدالة
print(is_prime(17))  # يجب أن يعرض True
print(is_prime(20))  # يجب أن يعرض False

}

console.log(greet("مبرمج"));

اضغط "تشغيل" لرؤية النتيجة
مثال: جمع الأعداد 
def calculate_sum(numbers):
    total = 0
    for num in numbers:
        total += num
    return total

# تجربة الدالة
my_numbers = [10, 20, 30, 40, 50]
result = calculate_sum(my_numbers)
print(f"مجموع الأعداد {my_numbers} = {result}")
